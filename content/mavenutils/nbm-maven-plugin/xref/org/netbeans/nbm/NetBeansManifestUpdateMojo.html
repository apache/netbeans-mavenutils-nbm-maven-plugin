63  /**
64   * Goal for generating NetBeans module system specific manifest entries, part of <code>nbm</code> lifecycle/packaging.
65   *
66   * In order to have the generated manifest picked up by the maven-jar-plugin,
67   * one shall add the following configuration snippet to maven-jar-plugin.
68   * 
69   * <pre>
70   *  {@code
71     <plugin>
72         <groupId>org.apache.maven.plugins</groupId>
73         <artifactId>maven-jar-plugin</artifactId>
74         <version>3.0.2</version>
75         <configuration>
76             <archive>
77                 <manifestFile>${project.build.outputDirectory}/META-INF/MANIFEST.MF</manifestFile>
78             </archive>
79         </configuration>
80     </plugin>
81   * }
82   * </pre>
83   *
84   * @author <a href="mailto:mkleint@codehaus.org">Milos Kleint</a>
85   */
86  @Mojo(name="manifest", 
87          defaultPhase= LifecyclePhase.PROCESS_CLASSES, 
88          requiresProject=true, 
89          threadSafe = true,
90          requiresDependencyResolution= ResolutionScope.RUNTIME )
91  public class NetBeansManifestUpdateMojo
92      extends AbstractNbmMojo

192     /**
193      * List of explicit module dependency declarations overriding the default specification dependency. Useful when depending on a range of major versions,
194      * depending on implementation version etc.
195      * <p>The format is:
196      * <pre>
197      * &lt;dependency&gt;
198      *    &lt;id&gt;groupId:artifactId&lt;/id&gt;
199      *    &lt;type&gt;spec|impl|loose&lt;/type&gt;
200      *    &lt;explicitValue&gt;the entire dependency token&lt;/explicitValue&gt;
201      * &lt;/dependency&gt;
202      * </pre>
203      * </p>
204      * <p>
205      * where <code>id</code> is composed of grouId and artifactId of a dependency defined in effective pom, separated by double colon. This is mandatory.</p>
206      * <p>
207      * Then there are 2 exclusively optional fields <code>type</code> and <code>explicitValue</code>, if both are defined <code>explicitValue</code> gets applied.
208      * </p>
209      * <p><code>type</code> values: <code>spec</code> means specification dependency.That's the default. 
210      * <code>impl</code> means implementation dependency, only the exact version match will satisfy the constraint. 
211      * <code>loose</code> means loose dependency, no requirement on version, the module just has to be present. Not very common option.
212      * 
213      * @since 3.8
214      */
215     @Parameter
216     private Dependency[] moduleDependencies;

219      * Deployment type of the module, allowed values are <code>normal</code>,<code>eager</code>,<code>autoload</code>,
220      * <code>disabled</code>.
221      * <p>
222      * <code>autoload</code> - Such a module is
223      * automatically enabled when some other module requires it and
224      * automatically disabled otherwise.</p>
225      *                     <p><code>eager</code> - This module type gets
226      * automatically enabled when all it's dependencies are
227      * satisfied. Disabled otherwise.</p>
228      *                     <p><code>normal</code> - This is the default
229      * value. This kind of module is enabled/disabled manually by
230      * the user. It installs enabled.</p>
231      *                     <p><code>disabled</code> - This kind of module is enabled/disabled manually by
232      * the user. It installs disabled. Since 3.11</p>
233      *
234      * For details, see <a href="http://bits.netbeans.org/dev/javadoc/org-openide-modules/org/openide/modules/doc-files/api.html#enablement">Netbeans Module system docs</a>
235      * 
236      * Since 3.14, for autoload and eager modules, we automatically set AutoUpdate-Show-In-Client manifest entry to false, if not defined already otherwise in the manifest.
237      * See issue <a href="http://jira.codehaus.org/browse/MNBMODULE-194">MNBMODULE-194</a>
238      * 
239      * @since 3.8 (3.14 in manifest goal)
240      */

242     protected String moduleType;

280     /**
281      * execute plugin
282      * @throws MojoExecutionException if an unexpected problem occurs
283      * @throws MojoFailureException if an expected problem occurs
284      */
285     public void execute()
286         throws MojoExecutionException, MojoFailureException
287 
288     {
289         //need to do this to chekc for javahelp on CP.
290         super.registerNbmAntTasks();
291         NetBeansModule module;
292         if ( descriptor != null && descriptor.exists() )
293         {
294             module = readModuleDescriptor( descriptor );
295             getLog().warn( "descriptor parameter is deprecated, use equivalent mojo parameters instead.");
296         }
297         else
298         {
299             module = createDefaultDescriptor( project, false );
300         }
301         
302         String mtype = moduleType;
303         //same moduleType related code in CreateNetBeansFileStructure.java
304         if ("normal".equals(mtype) && module.getModuleType() != null) {
305             mtype = module.getModuleType();
306             getLog().warn( "moduleType in module descriptor is deprecated, use the plugin's parameter moduleType");
307         }
308         if (!"normal".equals(mtype) && !"autoload".equals(mtype) && !"eager".equals(mtype) && !"disabled".equals(mtype)) {
309             getLog().error( "Only 'normal,autoload,eager,disabled' are allowed values in the moduleType parameter");
310         }
311         boolean autoload = "autoload".equals( mtype );
312         boolean eager = "eager".equals( mtype );
313         
314 
315         String moduleName = codeNameBase;
316         if (module.getCodeNameBase() != null) {
317             moduleName = module.getCodeNameBase();
318             getLog().warn( "codeNameBase in module descriptor is deprecated, use the plugin's parameter codeNameBase");
319         }
320         moduleName = moduleName.replaceAll( "-", "." );
321 //<!-- if a NetBeans specific manifest is defined, examine this one, otherwise the already included one.
322 // ignoring the case when some of the NetBeans attributes are already defined in the jar and more is included.
323         File specialManifest = sourceManifestFile;
324         File nbmManifest = ( module.getManifest() != null ? new File(
325             project.getBasedir(), module.getManifest() ) : null );
326         if ( nbmManifest != null && nbmManifest.exists() )
327         {
328             //deprecated, but if actually defined, will use it.
329             specialManifest = nbmManifest;
330         }
331         ExamineManifest examinator = new ExamineManifest( getLog() );
332         if ( specialManifest != null && specialManifest.exists() )
333         {
334             examinator.setManifestFile( specialManifest );
335             examinator.checkFile();
336         }
337         else
338         {
339 //            examinator.setJarFile( jarFile );
340         }
341 
342         getLog().info( "NBM Plugin generates manifest" );
343 
344         Manifest manifest = null;
345         if ( specialManifest != null && specialManifest.exists() )
346         {
347             Reader reader = null;
348             try
349             {
350                 reader = new InputStreamReader( new FileInputStream( specialManifest ) );
351                 manifest = new Manifest( reader );
352             }
353             catch ( IOException exc )
354             {
355                 manifest = new Manifest();
356                 getLog().warn( "Error reading manifest at " + specialManifest, exc );
357             }
358             catch ( ManifestException ex )
359             {
360                 getLog().warn( "Error reading manifest at " + specialManifest, ex );
361                 manifest = new Manifest();
362             }
363             finally
364             {
365                 IOUtil.close( reader );
366             }
367         }
368         else
369         {
370             manifest = new Manifest();
371         }
372         Date date = new Date();
373         String specVersion = AdaptNbVersion.adaptVersion( project.getVersion(),
374             AdaptNbVersion.TYPE_SPECIFICATION, date );
375         String implVersion = AdaptNbVersion.adaptVersion( project.getVersion(),
376             AdaptNbVersion.TYPE_IMPLEMENTATION, date );
377         Manifest.Section mainSection = manifest.getMainSection();
378         conditionallyAddAttribute( mainSection,
379             "OpenIDE-Module-Specification-Version", specVersion );
380         conditionallyAddAttribute( mainSection,
381             "OpenIDE-Module-Implementation-Version", implVersion );
382         if (autoload || eager) { //MNBMODULE-194
383             conditionallyAddAttribute( mainSection, "AutoUpdate-Show-In-Client", "false");
384         }
385         final String timestamp = createTimestamp( date );
386         conditionallyAddAttribute( mainSection, "OpenIDE-Module-Build-Version",
387             timestamp );
388         String projectCNB = conditionallyAddAttribute( mainSection, "OpenIDE-Module", moduleName );
389         String packagesValue;
390         if ( publicPackages != null && publicPackages.size() > 0 )
391         {
392             StringBuilder sb = new StringBuilder();
393             for ( String pub : publicPackages )
394             {
395                 if (pub == null) { //#MNBMODULE-237
396                     continue;
397                 }
398                 if ( pub.endsWith( ".**" ) )
399                 {
400                     // well, just sort of wrong value but accept
401                     sb.append( pub );
402                 }
403                 else if ( pub.endsWith( ".*" ) )
404                 {
405                     //multipackage value
406                     sb.append( pub ).append( "*" );
407                 }
408                 else
409                 {
410                     sb.append( pub ).append( ".*" );
411                 }
412                 sb.append( ", " );
413             }
414             if (sb.length() > 1) { //if only item is null, we have empty builder
415                 sb.setLength( sb.length() - 2 ); //cut the last 2 ", " characters
416                 packagesValue = sb.toString();
417             } else {
418                 // no packages available
419                 packagesValue = "-";
420             }
421         }
422         else
423         {
424             // no packages available
425             packagesValue = "-";
426         }
427         conditionallyAddAttribute( mainSection, "OpenIDE-Module-Public-Packages", packagesValue );
428 
429         //See http://www.netbeans.org/download/dev/javadoc/org-openide-modules/apichanges.html#split-of-openide-jar
430         conditionallyAddAttribute( mainSection, "OpenIDE-Module-Requires",
431             "org.openide.modules.ModuleFormat1" );
432 //        conditionallyAddAttribute(mainSection, "OpenIDE-Module-IDE-Dependencies", "IDE/1 > 3.40");
433         // localization items
434         if ( !examinator.isLocalized() )
435         {
436             conditionallyAddAttribute( mainSection,
437                 "OpenIDE-Module-Display-Category", project.getGroupId() );
438             conditionallyAddAttribute( mainSection, "OpenIDE-Module-Name",
439                 project.getName() );
440             conditionallyAddAttribute( mainSection,
441                 "OpenIDE-Module-Short-Description", shorten( project.getDescription() ) );
442             conditionallyAddAttribute( mainSection,
443                 "OpenIDE-Module-Long-Description", project.getDescription() );
444         }
445         getLog().debug( "module =" + module );
446         
447             DependencyNode treeroot = createDependencyTree( project, dependencyGraphBuilder, "compile" );
448             Map<Artifact, ExamineManifest> examinerCache = new HashMap<Artifact, ExamineManifest>();
449             @SuppressWarnings( "unchecked" )
450             List<Artifact> libArtifacts = getLibraryArtifacts( treeroot, module, project.getRuntimeArtifacts(),
451                 examinerCache, getLog(), useOSGiDependencies );
452             List<ModuleWrapper> moduleArtifacts = getModuleDependencyArtifacts( treeroot, module, moduleDependencies, project, examinerCache,
453                 libArtifacts, getLog(), useOSGiDependencies );
454             StringBuilder classPath = new StringBuilder();
455             StringBuilder mavenClassPath = new StringBuilder();
456             String dependencies = "";
457             String depSeparator = " ";
458 
459             for ( Artifact a : libArtifacts )
460             {
461                 if (classPath.length() > 0)
462                 {
463                     classPath.append(' ');
464                 }
465                 classPath.append(artifactToClassPathEntry( a, codeNameBase ));
466                 if ( mavenClassPath.length() > 0 )
467                 {
468                     mavenClassPath.append( ' ' );
469                 }
470                 mavenClassPath.append( a.getGroupId() ).append( ':' ).append( a.getArtifactId() ).append( ':' ).append( a.getBaseVersion() );
471                 if (a.getClassifier() != null) 
472                 {
473                     mavenClassPath.append(":").append(a.getClassifier());
474             }
475             }
476 
477             for ( ModuleWrapper wr : moduleArtifacts )
478             {
479                 if ( wr.transitive )
480                 {
481                     continue;
482                 }
483                 Dependency dep = wr.dependency;
484                 Artifact artifact = wr.artifact;
485                 ExamineManifest depExaminator = examinerCache.get( artifact );
486                 String type = dep.getType();
487                 String depToken = dep.getExplicitValue();
488                 if ( depToken == null )
489                 {
490                     if ( "loose".equals( type ) )
491                     {
492                         depToken = depExaminator.getModuleWithRelease();
493                     }
494                     else if ( "spec".equals( type ) )
495                     {
496                         depToken =
497                             depExaminator.getModuleWithRelease()
498                                 + " > "
499                                 + ( depExaminator.isNetBeansModule() ? depExaminator.getSpecVersion()
500                                                 : AdaptNbVersion.adaptVersion( depExaminator.getSpecVersion(),
501                                                                                AdaptNbVersion.TYPE_SPECIFICATION, date ) );
502                     }
503                     else if ( "impl".equals( type ) )
504                     {
505                         depToken =
506                             depExaminator.getModuleWithRelease()
507                                 + " = "
508                                 + ( depExaminator.isNetBeansModule() ? depExaminator.getImplVersion()
509                                                 : AdaptNbVersion.adaptVersion( depExaminator.getImplVersion(),
510                                                                                AdaptNbVersion.TYPE_IMPLEMENTATION, date ) );
511                     }
512                     else
513                     {
514                         throw new MojoExecutionException(
515                             "Wrong type of NetBeans dependency: " + type + " Allowed values are: loose, spec, impl." );
516                     }
517                 }
518                 if ( depToken == null )
519                 {
520                     //TODO report
521                     getLog().error(
522                         "Cannot properly resolve the NetBeans dependency for " + dep.getId() );
523                 }
524                 else
525                 {
526                     dependencies = dependencies + depSeparator + depToken;
527                     depSeparator = ", ";
528                 }
529             }
530             if ( !verifyRuntime.equalsIgnoreCase( SKIP ) )
531             {
532                 try
533                 {
534                     checkModuleClassPath( treeroot, libArtifacts, examinerCache, moduleArtifacts, projectCNB );
535                 }
536                 catch ( IOException ex )
537                 {
538                     throw new MojoExecutionException( "Error while checking runtime dependencies", ex );
539                 }
540             }
541 
542             if ( nbmJavahelpSource.exists() )
543             {
544                 String moduleJarName = stripVersionFromCodebaseName( moduleName ).replace( ".", "-" );
545                 classPath.append( " docs/").append( moduleJarName ).append( ".jar" );
546             }
547 
548             if ( classPath.length() > 0 )
549             {
550                 conditionallyAddAttribute( mainSection, "X-Class-Path", classPath.toString().trim() );
551             }
552             if ( mavenClassPath.length() > 0)
553             {
554                 conditionallyAddAttribute( mainSection, "Maven-Class-Path", mavenClassPath.toString() );
555             }
556             if ( dependencies.length() > 0 )
557             {
558                 conditionallyAddAttribute( mainSection, "OpenIDE-Module-Module-Dependencies", dependencies );
559             }
560 //            if ( librList.size() > 0 )
561 //            {
562 //                String list = "";
563 //                for ( int i = 0; i < librList.size(); i++ )
564 //                {
565 //                    list = list + " " + librList.get( i );
566 //                }
567 //                getLog().warn(
568 //                        "Some libraries could not be found in the dependency chain: " + list );
569 //            }
570         PrintWriter writer = null;
571         try
572         {
573             if ( !targetManifestFile.exists() )
574             {
575                 targetManifestFile.getParentFile().mkdirs();
576                 targetManifestFile.createNewFile();
577             }
578             writer = new PrintWriter( targetManifestFile, "UTF-8" ); //TODO really UTF-8??
579             manifest.write( writer );
580         }
581         catch ( IOException ex )
582         {
583             throw new MojoExecutionException( ex.getMessage(), ex );
584         }
585         finally
586         {
587             IOUtil.close( writer );
588         }
589     }

591     //MNBMODULE-137
592     static String artifactToClassPathEntry( Artifact a, String codenamebase )
593     {
594         return "ext/" + codenamebase + "/" + a.getGroupId().replace( '.', '-') + "/" + a.getArtifactId() + ( a.getClassifier() != null ? "-" + a.getClassifier() : "" ) + "." + a.getArtifactHandler().getExtension();
595     }

597     /**
598      * Create a timestamp for <code>OpenIDE-Module-Build-Version</code> manifest
599      * entry.
600      *
601      * It's created from the current time and formatted using a UTC timezone
602      * explicitly which makes created timestamp timezone-independent.
603      *
604      * @return timestamp represented as <code>201012292045</code>
605      */
606     private static String createTimestamp( Date date )
607     {
608         final SimpleDateFormat dateFormat = new SimpleDateFormat( "yyyyMMddHHmm" );
609         dateFormat.setTimeZone( TimeZone.getTimeZone( "UTC" ) );
610         final String timestamp = dateFormat.format( date );
611         return timestamp;
612     }

614     static String stripVersionFromCodebaseName( String cnb )
615     {
616         // it can happen the moduleName is in format org.milos/1
617         String base = cnb;
618         int index = base.indexOf( '/' );
619         if ( index > -1 )
620         {
621             base = base.substring( 0, index ).trim();
622         }
623         return base;
624     }

626     String conditionallyAddAttribute( Manifest.Section section, String key, String value )
627     {
628         Manifest.Attribute attr = section.getAttribute( key );
629         if ( attr == null )
630         {
631             attr = new Manifest.Attribute();
632             attr.setName( key );
633             attr.setValue( value != null ? value.replaceAll("\\s+", " ").trim() : "<undefined>" );
634             try
635             {
636                 section.addConfiguredAttribute( attr );
637             }
638             catch ( ManifestException ex )
639             {
640                 getLog().error( "Cannot update manifest (key=" + key + ")" );
641                 ex.printStackTrace();
642             }
643         }

652     static String shorten( String paragraph )
653     {

656             return null;

666     private void checkModuleClassPath( DependencyNode treeroot,
667         List<Artifact> libArtifacts,
668         Map<Artifact, ExamineManifest> examinerCache, List<ModuleWrapper> moduleArtifacts, String projectCodeNameBase )
669         throws IOException, MojoExecutionException, MojoFailureException
670     {
671         Set<String> deps = buildProjectDependencyClasses( project, libArtifacts );
672         deps.retainAll( allProjectClasses( project ) );
673 
674         Set<String> own = projectModuleOwnClasses( project, libArtifacts );
675         deps.removeAll( own );
676         CollectModuleLibrariesNodeVisitorlectModuleLibrariesNodeVisitor">CollectModuleLibrariesNodeVisitor visitor = new CollectModuleLibrariesNodeVisitor(
677             project.getRuntimeArtifacts(), examinerCache, getLog(), treeroot, useOSGiDependencies );
678         treeroot.accept( visitor );
679         Map<String, List<Artifact>> modules = visitor.getDeclaredArtifacts();
680         Map<Artifact, Set<String>> moduleAllClasses = new HashMap<Artifact, Set<String>>();
681 
682         for ( ModuleWrapper wr : moduleArtifacts )
683         {
684             if ( modules.containsKey( wr.artifact.getDependencyConflictId() ) )
685             {
686                 ExamineManifest man = examinerCache.get( wr.artifact );
687                 List<Artifact> arts = modules.get( wr.artifact.getDependencyConflictId() );
688                 Set<String>[] classes = visibleModuleClasses( arts, man, wr.dependency, projectCodeNameBase, false );
689                 deps.removeAll( classes[0] );
690                 moduleAllClasses.put( wr.artifact, classes[1] );
691             }
692         }
693 
694         //now we have the classes that are not in public packages of declared modules,
695         //but are being used
696         if ( !deps.isEmpty() )
697         {
698             Map<String, List<Artifact>> transmodules = visitor.getTransitiveArtifacts();
699             for ( ModuleWrapper wr : moduleArtifacts )
700             {
701                 if ( transmodules.containsKey( wr.artifact.getDependencyConflictId() ) )
702                 {
703                     ExamineManifest man = examinerCache.get( wr.artifact );
704                     List<Artifact> arts = transmodules.get( wr.artifact.getDependencyConflictId() );
705                     Set<String>[] classes = visibleModuleClasses( arts, man, wr.dependency, projectCodeNameBase, true );
706                     classes[0].retainAll( deps );
707                     if ( classes[0].size() > 0 )
708                     {
709                         String module = wr.osgi ? "OSGi bundle" : "module";
710                         getLog().error(
711                             "Project uses classes from transitive " + module + " " + wr.artifact.getId() + " which will not be accessible at runtime." );
712                         getLog().info( "    To fix the problem, add this module as direct dependency. For OSGi bundles that are supposed to be wrapped in NetBeans modules, use the useOSGiDependencies=false parameter");
713                         deps.removeAll( classes[0] );
714                     }
715                     classes[1].retainAll( deps );
716                     if ( classes[1].size() > 0 )
717                     {
718                         getLog().info( "Private classes referenced in transitive module: " + Arrays.toString( classes[1].toArray() ) );
719                         getLog().error(
720                             "Project depends on packages not accessible at runtime in transitive module " + wr.artifact.getId() + " which will not be accessible at runtime." );
721                         deps.removeAll( classes[1] );
722                     }
723                 }
724             }
725             for ( Map.Entry<Artifact, Set<String>> e : moduleAllClasses.entrySet() )
726             {
727                 List<String> strs = new ArrayList<String>( deps );
728                 if ( deps.removeAll( e.getValue() ) )
729                 {
730                     strs.retainAll( e.getValue() );
731                     getLog().info( "Private classes referenced in module: " + Arrays.toString( strs.toArray() ) );
732                     getLog().error( "Project depends on packages not accessible at runtime in module " + e.getKey().getId() );
733                 }
734             }
735             if ( verifyRuntime.equalsIgnoreCase( FAIL ) )
736             {
737                 if ( !deps.isEmpty() )
738                 {
739                     throw new MojoFailureException( "Uncategorized problems with NetBeans dependency verification (maybe MNBMODULE-102 or wrong maven dependency metadata). Supposedly external classes are used in the project's binaries but the classes are not found on classpath. Class usages: " + deps );
740                 }
741                 else
742                 {
743                     throw new MojoFailureException( "See above for failures in runtime NetBeans dependencies verification." );
744                 }
745             }
746         }
747     }

821     private Set<String>[] visibleModuleClasses( List<Artifact> moduleLibraries,
822         ExamineManifest manifest, Dependency dep, String projectCodeNameBase,
823         boolean transitive)
824         throws IOException, MojoFailureException
825     {
826         Set<String> moduleClasses = new HashSet<String>();
827         Set<String> visibleModuleClasses = new HashSet<String>();
828         DefaultClassAnalyzer analyzer = new DefaultClassAnalyzer();
829         String type = dep.getType();
830         if ( dep.getExplicitValue() != null )
831         {
832             if ( dep.getExplicitValue().contains( "=" ) )
833             {
834                 type = "impl";
835             }
836         }
837         if ( type == null || "loose".equals( type ) )
838         {
839             type = "spec";
840         }
841 
842         for ( Artifact lib : moduleLibraries )
843         {
844             URL url = lib.getFile().toURI().toURL();
845             moduleClasses.addAll( analyzer.analyze( url ) );
846         }
847 
848         if ( "spec".equals( type ) )
849         {
850             String cnb = stripVersionFromCodebaseName( projectCodeNameBase );
851             if ( !transitive && manifest.hasFriendPackages() && !manifest.getFriends().contains( cnb ) )
852             {
853                 String message = "Module has friend dependency on " + manifest.getModule() + " but is not listed as a friend.";
854                 if ( verifyRuntime.equalsIgnoreCase( FAIL ) )
855                 {
856                     throw new MojoFailureException( message );
857                 }
858                 else
859                 {
860                     getLog().warn( message );
861                 }
862             }
863             List<Pattern> compiled = createCompiledPatternList( manifest.getPackages() );
864             if ( useOSGiDependencies && manifest.isOsgiBundle() )
865             {
866                 // TODO how to extract the public packages in osgi bundles easily..
867                 compiled = Collections.singletonList( Pattern.compile( "(.+)" ) );
868             }
869             for ( String clazz : moduleClasses )
870             {
871                 for ( Pattern patt : compiled )
872                 {
873                     if ( patt.matcher( clazz ).matches() ) 
874                     {
875                         visibleModuleClasses.add( clazz );
876                         break;
877                     }
878                 }
879             }
880 
881         }
882         else if ( "impl".equals( type ) )
883         {
884             visibleModuleClasses.addAll( moduleClasses );
885         }
886         else
887         {
888             //HUH?
889             throw new MojoFailureException( "Wrong type of module dependency " + type );
890         }
891 
892         return new Set[]
893             {
894                 visibleModuleClasses,
895                 moduleClasses
896             };
897     }

899     static List<Pattern> createCompiledPatternList( List<String> packages )
900     {
901         List<Pattern> toRet = new ArrayList<Pattern>();
902         for ( String token : packages )
903         {
904             if ( token.endsWith( ".**" ) )
905             {
906                 String patt = "^" + Pattern.quote( token.substring( 0, token.length() - 2 ) ) + "(.+)";
907                 toRet.add( 0, Pattern.compile( patt ) );
908             }
909             else
910             {
911                 String patt = "^" + Pattern.quote( token.substring( 0, token.length() - 1 ) ) + "([^\\.]+)";
912                 toRet.add( Pattern.compile( patt ) );
913             }
914         }
915         return toRet;
916     }
